Day01

-- DDL之库的操作

-- 1.查看所有的库

show databases;


-- 2.创建库

-- 常见的字符集：UTF8 GBK GB2312 GB18030

create database mydb charset = UTF8;
create database if not exists mydb charset = UTF8;


-- 3.查看创建库的语句【字符集】

show create database mydb;


-- 4.使用库【切换库】

use mydb;


-- 5.删除库

drop database mydb;
drop database if exists mydb;
